报mysql.infoschema相关错，修改密码
DROP USER 'mysql.infoschema'@'localhost';
CREATE USER 'mysql.infoschema'@'localhost' IDENTIFIED BY 'password';
GRANT SELECT ON *.* TO `mysql.infoschema`@`localhost`;

stop和status失败，修改debian-sys-maint密码，先修改/etc/mysql/debian.cnf
drop user 'debian-sys-maint'@'localhost'
create user 'debian-sys-maint'@'localhost' identified by 'YES'
grant all privileges on *.* to 'debian-sys-maint'@'localhost' with grant option;
flush privileges;
